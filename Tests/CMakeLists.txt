set(TestsFolderName Tests)

        googletest
        SOURCE_DIR "${CMAKE_SOURCE_DIR}/ThirdParty/googletest"
        FIND_PACKAGE_ARGS NAMES GTest
)

include(GoogleTest)

# 允许使用add_test()添加测试，生成适用于ctest的测试配置。
enable_testing()
